160  void ConsoleEngine::renderLoop()
161  {
162  const unsigned int sleepms = 1000 / m_fps; // Number of ms to wait between each render
163  int64_t nextRender = 0;
164  while(m_rendering)
165  {
166  auto duration = chrono::system_clock::now().time_since_epoch();
167  int64_t millis = std::chrono::duration_cast<std::chrono::milliseconds>(duration).count();
168  if(millis >= nextRender)
169  {
170  nextRender = millis + sleepms;
171 
172  lock_guard<mutex> locker(m_guard); // Lock access to entire object
173  // Determine if it is needed to redraw
174  bool draw = m_needDraw;
175  if(!draw)
176  {
177  // Iterate over the sprite to see if there is one that need to be draw
178  for(int i = 0 ; i < LAYERS_NUMBER && !draw ; ++i)
179  {
180  for(const shared_ptr<Drawable> drawable : m_sprites[i])
181  {
182  // The sprite need to be drawn and is in the screen
183  if(drawable->needDraw()
184  && (isInsideScreen(drawable->getPosition(), m_size)
185  || isInsideScreen(drawable->getPosition() + drawable->m_size, m_size)))
186  {
187  // No need to continue to iterate if at least one sprite need redraw
188  draw = true;
189  break;
190  }
191  }
192  }
193  }
194 
195  // Draw if needed
196  if(draw)
197  {
198  clearBuffer(); // Clear the buffer
199 
200  // Ask each sprite to draw in the buffer
201  for(int i = 0 ; i < LAYERS_NUMBER ; ++i)
202  {
203  for(shared_ptr<Drawable> drawable : m_sprites[i])
204  drawable->draw(*this);
205  }
206 
207  // Print the buffer in the out stream
208  system(ENGINE_CLEAR_SCREEN_CMD);
209  for(int y = 0 ; y < m_size.y() ; ++y)
210  {
211  for(int x = 0 ; x < m_size.x() ; ++x)
212  cout << getPixel(Vec2<int>(x, y));
213  cout << endl;
214  }
215 
216  m_needDraw = false;
217  }
218  } // Release access
219 
220  // Test if there is new input
221  if(privateFunc::kbhit()) // Key pressed
222  {
223  char in = privateFunc::getch();
224  EventManager::getInstance()->pushEvent(new ConsoleKeyPressedEvent(in));
225  }
226 
227  this_thread::sleep_for(chrono::milliseconds(10)); // Little wait
228  }
229  }

310  {

318  // Read size

329 
330  m_pixels.clear();

332  while(c == '\n' || c == '\r')
333  file.get(c); // Go to next char of interest
334  for(int width = 1, height = 1 ; !file.eof() ; ++width)
335  {
336  if(c == '\n' || c == '\r')

338  // for debug

340  if(width != m_size.x() + 1)

344  return false; // Image size isn't correct
345  }
346  else
347  {
348  width = 0;
349  ++height;
350  }
351 
352  while(c == '\n' || c == '\r') // Needed on linux because line return often is \n\r
353  file.get(c); // Go to next char of interest

356 
357  if(height > m_size.y())
