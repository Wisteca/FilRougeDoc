172  void ConsoleEngine::renderLoop()
173  {
174  const unsigned int sleepms = 1000 / m_fps; // Number of ms to wait between each render
175  int64_t nextRender = 0;
176  while(m_rendering)
177  {
178  auto duration = chrono::system_clock::now().time_since_epoch();
179  int64_t millis = std::chrono::duration_cast<std::chrono::milliseconds>(duration).count();
180  if(millis >= nextRender)
181  {
182  nextRender = millis + sleepms;
183 
184  lock_guard<mutex> locker(m_guard); // Lock access to entire object
185  // Determine if it is needed to redraw
186  bool draw = m_needDraw;
187  if(!m_needDraw)
188  {
189  // Iterate over the sprite to see if there is one that need to be draw
190  for(int i = 0 ; i < LAYERS_NUMBER && !draw ; ++i)
191  {
192  for(const Sprite* sprite : m_sprites[i])
193  {
194  // The sprite need to be draw and is in the screen
195  if(sprite->needDraw()
196  && (isInsideScreen(sprite->getPosition(), m_size)
197  || isInsideScreen(sprite->getPosition() + sprite->m_size, m_size)))
198  {
199  // No need to continue to iterate if at least one sprite need redraw
200  draw = true;
201  break;
202  }
203  }
204  }
205  }
206 
207  // Draw if needed
208  if(draw)
209  {
210  clearBuffer(); // Clear the buffer
211 
212  // Ask each sprite to draw in the buffer
213  for(int i = 0 ; i < LAYERS_NUMBER ; ++i)
214  {
215  for(Sprite* sprite : m_sprites[i])
216  sprite->draw(*this);
217  }
218 
219  // Print the buffer in the out stream
220  system(ENGINE_CLEAR_SCREEN_CMD);
221  for(int y = 0 ; y < m_size.y() ; ++y)
222  {
223  for(int x = 0 ; x < m_size.x() ; ++x)
224  cout << getPixel(Vec2<int>(x, y));
225  cout << endl;
226  }
227 
228  m_needDraw = false;
229  }
230  } // Release access
231 
232  // Test if there is new input
233  if(privateFunc::kbhit()) // Key pressed
234  {
235  char in = privateFunc::getch();
236  EventManager::getInstance()->pushEvent(new ConsoleKeyPressedEvent(in));
237  }
238 
239  this_thread::sleep_for(chrono::milliseconds(1)); // Little wait
240  }
