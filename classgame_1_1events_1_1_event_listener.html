Detailed Description

This is the interface each class who wants to listen to events must implement.

This is a really simple oriented objects events system. Here a little example to demonstrate how it works :

#include <EventManager.hpp>
// ExampleListener implements EventListener as private because there is no need for ExampleListener
// users to know that it listens to events.
class ExampleListener : private EventListener
{
public:
inline ExampleListener()
{
// Add the listener in the EventManager list at object construction
EventManager::getInstance()->addListener(this);
}
inline virtual ~ExampleListener()
{
// Don't forget to remove the listener from the list at destruction !
EventManager::getInstance()->removeListener(this);
}
inline void onEvent(const shared_ptr<Event> event) override
{
// Implement the onEvent method that will be called for each event.
Logger::log() << info << "Processing event " << event->getEventTypeName() << endlog;
switch(event->getEventType())
{
case EventType::ATTACK_EVENT :
Logger::log() << info << "Attack event !" << endlog; // Do something in case of attack event, ...
break;
case EventType::MOVE_EVENT :
Logger::log() << info << "Move event !" << endlog;
break;
}
}
};
