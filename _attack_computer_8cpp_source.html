52  return;
53  }

55  // ----- 1. Bonus given by the attack skill -----

87  void AttackComputer::targetDodge()
88  {
89  if(!m_attacker || !m_target)
90  {
91  Logger::log() << severe << "Please configure an attacker and a target before call targetDodge() !" << endlog;
92  return;
93  }
94 
95  // Determine which dodge takes effect
96  int skDodge = 0;
97  switch(m_attackType)
98  {
99  case AttackType::MAGICAL_ATTACK :
100  skDodge = m_target->getSkillsSet().computeActualValue(SkillType::DODGE_MAGIC);
101  break;
102  case AttackType::WEAPON_ATTACK :
103  case AttackType::HANDS_ATTACK :
104  skDodge = m_target->getSkillsSet().computeActualValue(SkillType::DODGE_PHYSIC);
105  break;
106  }
107  skDodge += 1; // Avoid division by zero
108 
109  // Calcul a random value between 0 and the dodge value of target
110  srand(time(nullptr));
111  int dodgeLuck = rand() % skDodge;
112  Logger::log() << debug << "Random dodge luck is " << dodgeLuck << " of " << skDodge << "." << endlog;
113 
114  // Compare the dodge luck to the attack skill of attacker
115  int dodgeResult = dodgeLuck - getAttackerAttackSkill();
116  int dodgeScale = dodgeLuck * attack_rules::DODGE_SCALE;
117  Logger::log() << "Little success if result between " << -dodgeScale << " and 0, total success if > 0." << endlog;
118  Logger::log() << "dodgeResult = dodgeLuck - attackerAttackSkill = " << dodgeLuck << " - " << getAttackerAttackSkill() << " = " << dodgeResult << " => ";
119  // if dodgeResult is positive, the dodge is a success and no damage comes to the target
120  // if dodgeResult is within DODGE_SCALE % of intial dodgeLuck, the damages are proportionally decreased compared to dodgeResult
121  // if dodgeResult is too negative, the dodge failed
122  if(dodgeResult > -dodgeScale && dodgeResult < 0) // Little success
123  {
124  int diff = dodgeScale + dodgeResult;
125  float successV = (float) diff / dodgeScale;
126  m_magicalDamage *= successV;
127  m_physicalDamage *= successV;
128 
129  Logger::log() << "Little success" << endlog;
130  Logger::log() << "Difference between dodgeScale and dodgeResult is " << diff << ", that means a dodge success of " <<
131  successV * 100 << "% damages are multiplied by this value." << endlog;
132  }
133  else if(dodgeResult >= 0) // Total success
134  {
135  m_magicalDamage = m_physicalDamage = 0;
136  Logger::log() << "Total success, damages are 0." << endlog;
137  }
138  else
139  Logger::log() << "Dodge failed, damages doesn't change." << endlog;
140  }
